# requirements.txt

# This file lists every external Python package the project depends on.
# You install them all at once with:  pip install -r requirements.txt
#
# Phase 1 uses ONLY Python's built-in standard library (no external
# packages needed yet) — that is intentional, so you can confirm your
# environment works before adding any dependencies.
#
# Phase 2 adds our first external dependency: customtkinter, a library
# that gives Python's built-in Tkinter GUI toolkit a modern look
# (dark mode, rounded buttons, clean fonts).
customtkinter==6.0.0

# Real AI reasoning via Google's Gemini API (switched from Anthropic's
# Claude — Gemini has a genuinely free tier via Google AI Studio, no
# credit card required, which matters while you're still testing), plus
# python-dotenv to safely load the API key from a local .env file
# instead of hardcoding it in source code.
google-genai
python-dotenv

# Phase 4 adds voice: pyttsx3 for offline text-to-speech, and
# SpeechRecognition + PyAudio for microphone capture and transcription.
pyttsx3
SpeechRecognition==3.17.0
pyaudio

# Phase 5 (automation) uses only Python's standard library (subprocess,
# pathlib, webbrowser, os) — no new dependencies needed. Tool-use itself
# is a feature of the google-genai package installed above.

# Phase 8 adds window management (list/focus/minimize/maximize/close).
# NOTE: pygetwindow has full support on Windows, partial support on
# Linux (requires an EWMH-compliant window manager), and very limited
# support on macOS. JARVIS still runs fine everywhere — window control
# specifically just reports itself as unavailable where unsupported.
pygetwindow
